create table country(
id int IDENTITY(1,1) PRIMARY KEY,
name varchar(50) NOT NULL,
popuation int,
area int)

create table city(
id int IDENTITY(1,1) PRIMARY KEY,
cityname varchar(50) NOT NULL,
country_id int,
popuation int,
rating int)
